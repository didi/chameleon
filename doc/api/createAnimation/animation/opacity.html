animation.opacity()

设置透明度

参数

返回值 类型 说明
number value number 高度值

返回值

返回值 类型 说明
animation animation 动画实例

示例

<template>
  <view>
    <view class="block" c-animation="{{animationData}}" c-bind:click="handleClick">
      <text>请点击我</text>
    </view>
  </view>
</template>
<script>
  import { createAnimation } from "@didi/chameleon-api";
  const animation = createAnimation();
  class Index {
    data = {
      animationData: null,
    }
    methods = {
      handleClick() {
        this.animationData = animation
          .opacity(0.1).step({duration: 1000})
          .export();
      }
    }
  }
  export default new Index();
</script>
<style scoped>
.block {
  position: absolute;
  width: 200cpx;
  height: 200cpx;
  background-color: #E3EDCD;
}
</style>
<script cml-type="json">
{
    "base": {
        "usingComponents": {}
    }    
}
</script>
